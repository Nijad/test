program TypeMismatch
{
    void test() {
        int x;
        double y = 3.14;
        x = y;  // خطأ دلالي: فقدان دقة (تحذير)

        boolean flag;
        flag = 5;  // خطأ دلالي: لا يمكن تعيين int إلى boolean

        int z;
        z = false;  // خطأ دلالي: لا يمكن تعيين boolean إلى int 
    }
}

program InvalidOperations
{
    void test() {
        int x = 5;
        boolean b = true;

        int result = x + b;  // خطأ دلالي: لا يمكن جمع int و boolean

        boolean b2 = x && b;  // خطأ دلالي: لا يمكن استخدام && مع int
    }
}
