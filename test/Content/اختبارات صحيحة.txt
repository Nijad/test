اختبارات حالات متكاملة
// اختبار 60: برنامج متكامل
program Calculator {
    // هياكل
    struct Vector {
        double x;
        double y;
    }
    
    // متغيرات عالمية
    double PI = 3.14159;
    
    // دوال
    function double dotProduct(Vector v1, Vector v2) {
        return v1.x * v2.x + v1.y * v2.y;
    }
    
    function void main() {
        Vector a, b;
        a.x = 1.0;
        a.y = 2.0;
        b.x = 3.0;
        b.y = 4.0;
        
        double result = dotProduct(a, b);
        
        if (result > 10) {
            return;
        } else {
            while (result < 100) {
                result = result * 2;
            }
        }
    }
}
